Python - Audio Processing 

import requests
from pathlib import Path
import subprocess

def download_audio(url, output_path):
    """
    Download audio from URL with progress tracking
    """
    try:
        response = requests.get(url, stream=True)
        response.raise_for_status()
        
        total_size = int(response.headers.get('content-length', 0))
        downloaded = 0
        
        with open(output_path, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                if chunk:
                    f.write(chunk)
                    downloaded += len(chunk)
                    progress = (downloaded / total_size) * 100
                    print(f'Download progress: {progress:.1f}%')
        
        return True
    except Exception as e:
        print(f'Error downloading audio: {e}')
        return False

# Usage example
if __name__ == "__main__":
    url = "https://example.com/audio.mp3"
    output = "downloaded_audio.mp3"
    success = download_audio(url, output)
    print(f'Download {"successful" if success else "failed"}')

C# - File Management 

using System;
using System.IO;
using System.Linq;
using System.Collections.Generic;

namespace FileManager
{
    public class FileOrganizer
    {
        public static void OrganizeByExtension(string sourcePath)
        {
            try
            {
                var files = Directory.GetFiles(sourcePath);
                var organizedCount = 0;

                foreach (var file in files)
                {
                    var extension = Path.GetExtension(file).ToLower();
                    var targetFolder = Path.Combine(sourcePath, extension.TrimStart('.'));
                    
                    if (!Directory.Exists(targetFolder))
                    {
                        Directory.CreateDirectory(targetFolder);
                    }

                    var fileName = Path.GetFileName(file);
                    var targetPath = Path.Combine(targetFolder, fileName);
                    
                    // Handle duplicate filenames
                    if (File.Exists(targetPath))
                    {
                        var nameWithoutExt = Path.GetFileNameWithoutExtension(file);
                        var newName = $"{nameWithoutExt}_{DateTime.Now.Ticks}{extension}";
                        targetPath = Path.Combine(targetFolder, newName);
                    }

                    File.Move(file, targetPath);
                    organizedCount++;
                    Console.WriteLine($"Moved: {fileName} -> {targetFolder}");
                }

                Console.WriteLine($"Organized {organizedCount} files successfully!");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Error organizing files: {ex.Message}");
            }
        }
    }
}
